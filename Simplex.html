import numpy as np import pandas as pd from js import document def run_simplex(evt=None): attackerCost = float(document.getElementById("attCost").value) defenderCost = float(document.getElementById("defCost").value) AP = float(document.getElementById("ap").value) DP = float(document.getElementById("dp").value) alphaA, alphaD = 0.4273, 0.0355 betaA, betaD = 0.2464, 0.9223 # Solve simple 2x2 system try: solution = np.linalg.solve(np.array([[alphaA, alphaD],[betaA, betaD]]), np.array([AP, DP])) solution = np.clip(solution, 0, None) except np.linalg.LinAlgError: solution = np.array([0,0]) # enforce team size <= 10 if solution.sum() > 10: solution = solution * 10 / solution.sum() contributions = solution * np.array([attackerCost, defenderCost]) total_cost = contributions.sum() df = pd.DataFrame({ "Type": ["Attacker", "Defender"], "Ratio": np.round(solution, 2), "Costs (€)": [f"€{v:,.2f}" for v in contributions] }) df = pd.concat([df, pd.DataFrame({"Type":["Total Cost"], "Ratio":["-"], "Costs (€)":[f"€{total_cost:,.2f}"]})], ignore_index=True) document.getElementById("result").innerHTML = df.to_html(index=False) document.getElementById("run").addEventListener("click", run_simplex)
